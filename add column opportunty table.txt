CREATE TABLE opportunity (
items_id INTEGER,
items_created TIMESTAMPTZ NOT NULL DEFAULT NOW (),
items_modified TIMESTAMPTZ NOT NULL DEFAULT NOW (),
items_lead_userId VARCHAR(50),
items_lead_fiId VARCHAR(50),
items_lead_name VARCHAR(50),
items_currentBalanceInPennies INTEGER,
items_dueBy TIMESTAMPTZ NOT NULL DEFAULT NOW (),
items_retaining INTEGER NOT NULL,
items_loanType VARCHAR(50),
items_tags VARCHAR(50),
items_location_city VARCHAR(50),
items_location_state VARCHAR(50),
items_location_street VARCHAR(50),
items_rate INTEGER NOT NULL,
items_term INTEGER NOT NULL,
items_description VARCHAR(50),
items_amortization INTEGER NOT NULL,
items_grading VARCHAR(50),
items_leadFee_serviceFee INTEGER NOT NULL,	
items_leadFee_flatFeeInPennies INTEGER NOT NULL,
items_status VARCHAR (50),
items_percentFunded INTEGER NOT NULL,
items_percentCommitted INTEGER NOT NULL,
items_contractId VARCHAR (50),
items_resourceType VARCHAR(50),
count INTEGER NOT NULL
);
